def avg(model, x, y, eq):
    pred = model.predict(x)
    acc = [1 if eq(e, s) else 0 for (e,s) in zip(pred, y)]
    return np.average(acc)


def filter_diff(x):
    rp = x['RiskPerformance']
    dis = len(set(rp))
    return dis == 1


def get_data():
    read = pd.read_csv("heloc_dataset_v1.csv")
    data_columns = [f for f in read.columns if f != 'RiskPerformance']
    dc2 = [f for f in read.columns if f != 'RiskPerformance' and f != 'ExternalRiskEstimate']
    grouped = read.groupby(data_columns)
    filtered = grouped.filter(lambda x: True)
    return filtered['RiskPerformance'], filtered[data_columns]


def prep_data():
    res, data = get_data()
    res = [1 if t == "Good" else 0 for t in res]
    should_use = use_columns()
    data = data[[c for c in data.columns if should_use[c] == 1]]
    for c in data.columns:
          data[c] = [-9 if t == -7 or t == -8 or t == -9 else t for t in data[c]]
    orig_data = data
    return orig_data, np.array(res), pd.DataFrame(data, columns=orig_data.columns) #MinMaxScaler().fit_transform(data)

Ponownie oba typy wykresów są bardzo podobne, choć tym razem nie są identyczne. W tym modelu wykresy nie są idealnie monotoniczne, lecz generalny trend wykresu jest zgodny z oczekiwaną monotonicznością.

Dla porówniania między modelami zestawmy wykresy ze sobą dla większej czytelności.

W porównaniu wykresów między dwoma modelami widać że wykresy drugiego modelu są bardziej gładkie, nie wystepują w nich tak gwałtowne skoki jak na wykresach NetFractionRevolvingBurden i PercentTradesNeverDelq modelu XGBoost. Ogólnie rzecz ujmując wykresy są bardzo podobne choć dla zmiennych AverageMInFile i PercentTradesNeverDelq wykresy XGBoost przedstawiają zupełnie inne zachowanie modelu dla wartość tych zmiennych poniżej zera niż w modelu Random Forest
