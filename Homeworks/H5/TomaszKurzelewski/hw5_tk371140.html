def avg(model, x, y, eq):
    pred = model.predict(x)
    acc = [1 if eq(e, s) else 0 for (e,s) in zip(pred, y)]
    return np.average(acc)

def filter_diff(x):
    rp = x['RiskPerformance']
    dis = len(set(rp))
    return dis == 1

def get_data():
    read = pd.read_csv("heloc_dataset_v1.csv")
    data_columns = [f for f in read.columns if f != 'RiskPerformance']
    grouped = read.groupby(data_columns)
    filtered = grouped.filter(lambda x: True)
    return filtered['RiskPerformance'], filtered[data_columns]

def prep_data():
    res, data = get_data()
    res = [1 if t == "Good" else 0 for t in res]
    for c in data.columns:
          data[c] = [-9 if t == -7 or t == -8 or t == -9 else t for t in data[c]]
    orig_data = data
    return orig_data, np.array(res), pd.DataFrame(data, columns=orig_data.columns)

    for i in imps:
        index = int(i.feature[1:])
        l[index] = i.weight
    allData.append(list([l[i] if i in l else 0 for i in range(0, len(varNames))]))
# take 7 with greatest sum of weights

for d in allData:
    for i, v in enumerate(d):

sorted.reverse()

allData = [[d[s] for s in sorted] for d in allData]
varNames = [varNames[s] for s in sorted]

Zauwazmy e w modelu uywajcym zmiennej ExternalRiskEstimate jest jedna istotna zmienna, podczas gdy w powyszych trzech modelach nieuywajcych tej zmennej s zawsze conajmniej trzy zmienne odgrywajce znaczc rol, takie jak NetFractionRevolvingBurden, AverageMInFile czy PercentTradesNeverDelq. Wynika z tego e usunicie tej zmiennej prowadzi do lepiej wyjanialnego modelu decyzji kredytowych. Wiadomo e zewntrzny scoring zawsze odgrywa wan rol, podczas gdy dla nas istotniejsza jest informacja ktre konkretnie parametry sytuacji finansowej klienta s brane pod uwag.

Zmienne pojawiajce si jako istotne w powyszych modelach pojawiay si rwnie w poprzednich pracach jako istotne, co potwierdza poprawno wyjanienia. Pomimo podobiestw midzy modelami, takich jak e zmienn z najwysz wag we wszystkich modelach jest NetFractionRevolvingBurden, wystpuj te rnice midzy nimi. W modelu XGBoost drug z najwaniejszych zmiennych jest NumBank2NatlTradesWHighUtilization, podczas gdy w modelach lasu losowego ta zmienna jest odpowiednio na 6-tej.
