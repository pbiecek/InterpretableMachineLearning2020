    feature_names = features.columns

    shap.force_plot(
        explainer.expected_value[1],
        shap_values[index],
        feature_names=feature_names, 
        matplotlib=True, show=True)

def most_important(n: int, arr: np.ndarray):
    return tuple(sorted(arr.argsort()[-n:]))

def find_two_examples(k_important: int, explainer, features: pd.DataFrame):
    data = features.iloc[0:100, :]
    shap_values = explainer.shap_values(data)[1]
    for (id1, data1), (id2, data2) in product(enumerate(shap_values), enumerate(shap_values)):
        data1 = most_important(k_important, np.abs(data1))
        data2 = most_important(k_important, np.abs(data2))
        if len(set(data1) & set(data2)) == 0:
            display(X_test.iloc[id1:(id1+1), :])
            display(X_test.iloc[id2:(id2+1), :])
            feature_names = features.columns
            shap.force_plot(
                explainer.expected_value[1],
                shap_values[id1],
                feature_names=feature_names, 
                matplotlib=True, show=True)
            shap.force_plot(
                explainer.expected_value[1],
                shap_values[id2],
                feature_names=feature_names, 
                matplotlib=True, show=True)
            break

def find_two_examples(explainer1, explainer2, features: pd.DataFrame):
    data = features.iloc[0:100, :]
    shap_values1 = explainer1.shap_values(data)[1]
    shap_values2 = explainer2.shap_values(data)
    best_diff = 10 ** 10
    best_id1 = 0
    best_id2 = 0
    for (id1, data1), (id2, data2) in zip(enumerate(shap_values1), enumerate(shap_values2)):
        diff = np.sum(np.abs(data1 - data2))
        if id1 != id2 and diff < best_diff:
            beast_diff = best_diff
            best_id1 = id1
            best_id2 = id2

    display(X_test.iloc[best_id1:(best_id1+1), :])
    feature_names = features.columns
    shap.force_plot(
        explainer1.expected_value[1],
        shap_values1[best_id1],
        feature_names=feature_names, 
        matplotlib=True, show=True)
    shap.force_plot(
        explainer2.expected_value,
        shap_values2[best_id2],
        feature_names=feature_names, 
        matplotlib=True, show=True)
